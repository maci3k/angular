AngularJS nie byłby bez dyrektyw tym samym frameworkiem. To właśnie dyrektywy są absolutną przewagą AngularJS. Tak jak w tytule – użycie dyrektyw, to HTML na sterydach. Dzięki nim jesteśmy w stanie nauczyć nasz HTML nowych sztuczek.

Krótkie demo tworzenia własnej dyrektywy.

plik navBar.directive.js 

(function ()
{

    'use strict';

    function navBar()
    {
        return {
            restrict: 'ACE',
            controller: 'NavBar',
            controllerAs: 'nav',
            templateUrl: 'directive/navBar.tpl.html'
        };
    }

    angular.module('helloAngular').directive('navBar', [ navBar ]);


})();

plik navBar.controller.js 

(function ()
{
    'use strict';
    function NavBarController()
    {
        var ctrl = this;
    }


    angular.module('helloAngular')
            .controller('NavBar', [ NavBarController ]);

})();

Dodatkowo z plikuindex.html skopiuj element 'nav' z zawartością do pliku navBar.tpl.html

Nie zapomnij dodać skryptów (controllera oraz dyrektywy) na końcu pliku index.html.
